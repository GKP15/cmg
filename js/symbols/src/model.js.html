 59 /** 
 60  * mainloop wird alle ONE_FRAME_TIME aufgerufen

 64         updateGame();

 66     };

 92  /**
 93   *  ueberprueft ob sich zwei spieler auf dem selben laengen und breitengrad befinden, wenn ja wird eine "melodie" gespielt
 94  *  @param playerOne: spieler dessen position verglichen werden soll
 95  *  @param playerTwo: spieler dessen position verglichen werden soll
 96  */
 97 function checkCollision(playerOne,playerTwo) {
 98         if (playerOne.getPosition().lat() == playerTwo.getPosition().lat() 
 99         		&& playerOne.getPosition().lng() == playerTwo.getPosition().lng()) {
100                 	ouch.play();
101         }
102 }

104 /**
105  * eine funktion die den geist bewegen soll
106  */
107 function goGhost() {
108         if(changeDir == 0) {
109 			number = Math.floor( Math.random() * 4 );
110 			changeDir = 1 + Math.floor(Math.random() * 4);
111 		}
112 		changeDir--;
113         switch (number) {
114                 case 0: //left
115                         moveLeft(ghost);
116                         break;
117                 case 1: //right
118                         moveRight(ghost);
119                         break;
120                 case 2: //up
121                         moveUp(ghost);
122                         break;
123                 case 3: //down
124                         moveDown(ghost);
125                         break;
126         }
127 }

129 /**
130  *  wenn pucman noch nicht gesetzt wurde wird er auf die Mitte der karte gesetzt
131  * ansonsten wird je nach eingabe eine funktion aufgerufen, die pucman in die
132  * entsprechende richtung bewegt
133  *  @param key: taste die zuletzt gedrueckt wurde, w,a,s,d entsprechen den pfeiltasten alles andere wird auf null gesetzt
134  */
135 function gopucman(key) {
136 	if (key != null) {
137 		if (pucman == null) {
138 			pucman = new google.maps.Marker({
139 			            position: map.getCenter(),
140 			            map: map,
141 			            title: 'HERO',
142 			            icon: pucmanIconRight,
143 			            draggable: false
144 			});
145 			open = true;
146 			infoWindow.close();
147 		}
148 		//welche Taste wurde gedrueckt?
149 			//mit keycode funktionieren die pfeiltasten, funktioniert aber nicht im firefox
150 			//var pressedKey = key.keyCode; 
151 			//funktioniert bei firefox
152 		var pressedKey = key.charCode;
153 		//jenachdem was gedrueckt wurde wird die entsprechende funktion aufgerufen
154 		// ist das jetzt noch WASD oder cursortasten?
155 		switch (pressedKey) {
156 			case 97: // a
157 			    moveLeft(pucman);
158 			    break;
159 			case 100: // d
160 			    moveRight(pucman);
161 			    break;
162 			case 119: // w
163 			    moveUp(pucman);
164 			    break;
165 			case 115: // s
166 			    moveDown(pucman);
167 			    break;
168 			default: 
169 				// falls andere Taste gedrueckt, passiert nichts
170 				break;
171 		}
172 	}
173 }

176 /**
177  * bewegung nach links
178  * laengen und breitengrad werte werden entsprechend veraendert
179  * @param player: zu bewegendes spielObjekt(pucman oder geist)
180  */
181 function moveLeft(player){
182 		if (player == pucman) {
183 		        newPosition = new google.maps.LatLng(player.getPosition().lat(), player.getPosition().lng() - step);
184 		        if (open) {
185 		                pucman.setIcon(pucmanIconClose);
186 		                open = false;
187 		        } else {
188 		                pucman.setIcon(pucmanIconLeft);
189 		                open = true;
190 		        }
191 		}
192 		if (player == ghost) {
193 		        newGhostPosition = new google.maps.LatLng(player.getPosition().lat(), player.getPosition().lng() - step);
194 		}
195 }

375 /**
376  * sucht geo-koordinaten zu eingegebener Adresse
377  */
378 function codeAddress() {
379         //speichert die Geodaten des Eingebenen Ortes
380         var home;
381         var address = document.getElementById('address').value;
382 		if(address == document.getElementById('address').defaultValue) address = 'Leipzig';
383         geocoder.geocode( { 'address': address}, function(results, status) {
384                 if (status == google.maps.GeocoderStatus.OK) {
385                         map.setCenter(results[0].geometry.location);
386                         //setzt den Zoomfaktor nach der Eingabe eines Ortes
387                         zoom = 18;
388                         map.setZoom(zoom);
389                         //speichert die geodaten des ergebnisses in die variable home
390                         home = google.maps.LatLng(results[0].geometry.location);
391                         // Set the info window's content and position.
392                         infoWindow.setContent("Wenn du eine beliebige Taste drückst, <br> kannst den Geist mit awds steuern.");
393                         //infoWindow.setPosition(home);
394                         infoWindow.setPosition(map.getCenter());
395                         infoWindow.open(map);
396                 } else {
397                         alert('Geocode was not successful for the following reason: ' + status);
398                 }
399                 audio.play();
400                 //soabld die geosuche abgeschlossen ist kann man pucman steuern.
401                 google.maps.event.addDomListener(
402                 		document, 'keypress', function(pressedKey){key = pressedKey;}
403                 );
404                 newGhostPosition = new google.maps.LatLng(map.getCenter().lat() - step, map.getCenter().lng() + step);
405                 ghost.setPosition(newGhostPosition);
406                 ghost.setMap(map);
407 				map.set('disableDefaultUI', true);
408 				map.set('scrollwheel', false);
409 				map.set('draggable', false);
410 				map.set('disableDoubleClickZoom', true);
411         });
412 }
